User.php

            
    <?php

    class User
    {
        public function isLoggedIn()
        {
            if (isset($_SESSION['loggedin']){
                return $_SESSION['loggedin'];
            } elseif (isset($_COOKIE['Loggedin']){
                return $_COOKIE['Loggedin'];
            }
        }
    }
